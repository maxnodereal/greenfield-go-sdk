Overview ▾

Package runenames provides rune names from the Unicode Character Database. For example, the name for '\u0100' is "LATIN CAPITAL LETTER A WITH MACRON".

See https://www.unicode.org/Public/UCD/latest/ucd/UnicodeData.txt

▾ Example

Code:

runes := []rune{
    -1,
    '\U00000000',
    '\U0000001f',
    '\U00000020',
    '\U00000021',
    '\U00000041',
    '\U0000007e',
    '\U0000007f',
    '\U00000080',
    '\U000000e0',

    '\U0000037f',
    '\U00000380',
    '\U00000381',
    '\U00000382',
    '\U00000383',
    '\U00000384',
    '\U00000385',
    '\U00000386',
    '\U000007c0',

    '\U00002603',
    '\U000033ff',
    '\U00003400',
    '\U00003401',
    '\U00003402',
    '\U00004dc0',

    '\U00009fd5',
    '\U00009fff',
    '\U0000a000',
    0xdc00, // '\U0000dc00' (Low Surrogate) is an invalid Go literal.
    '\U0000f800',
    '\U0000fffc',
    '\U0000fffd',
    '\U0000fffe',
    '\U0000ffff',

    '\U00010000',
    '\U0001f574',
    '\U0002fa1d',
    '\U0002fa1e',
    '\U000e0100',
    '\U000e01ef',
    '\U000e01f0',
    '\U00100000',
    '\U0010fffd',
    '\U0010fffe',
    '\U0010ffff',
}

for _, r := range runes {
    fmt.Printf("%08x %q\n", r, runenames.Name(r))
}
