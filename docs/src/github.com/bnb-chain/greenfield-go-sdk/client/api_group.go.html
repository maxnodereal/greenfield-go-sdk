   250  	const (
   251  		MaximumGetGroupListLimit  = 1000
   252  		MaximumGetGroupListOffset = 100000
   253  		DefaultGetGroupListLimit  = 50
   254  	)
   255  
   256  	if name == "" {
   257  		return types.ListGroupsResult{}, nil
   258  	}
   259  
   260  	if prefix == "" {
   261  		return types.ListGroupsResult{}, nil
   262  	}
   263  
   264  	if opts.Limit < 0 {
   265  		return types.ListGroupsResult{}, nil
   266  	} else if opts.Limit > 1000 {
   267  		opts.Limit = MaximumGetGroupListLimit
   268  	} else if opts.Limit == 0 {
   269  		opts.Limit = DefaultGetGroupListLimit
   270  	}
   271  
   272  	if opts.Offset < 0 || opts.Offset > MaximumGetGroupListOffset {
   273  		return types.ListGroupsResult{}, nil
   274  	}
   275  
   276  	if opts.SourceType != "" {
   277  		if _, ok := storageTypes.SourceType_value[opts.SourceType]; !ok {
   278  			return types.ListGroupsResult{}, nil
   279  		}
   280  	}
   281  
   282  	params := url.Values{}
   283  	params.Set("group-query", "")
   284  	params.Set("name", name)
   285  	params.Set("prefix", prefix)
   286  	params.Set("source-type", opts.SourceType)
   287  	params.Set("limit", strconv.FormatInt(opts.Limit, 10))
   288  	params.Set("offset", strconv.FormatInt(opts.Offset, 10))
   289  	reqMeta := requestMeta{
   290  		urlValues:     params,
   291  		contentSHA256: types.EmptyStringSHA256,
   292  	}
   293  
   294  	sendOpt := sendOptions{
   295  		method:           http.MethodGet,
   296  		disableCloseBody: true,
   297  	}
   298  
   299  	endpoint, err := c.getInServiceSP()
   300  	if err != nil {
   301  		log.Error().Msg(fmt.Sprintf("get in-service SP fail %s", err.Error()))
   302  		return types.ListGroupsResult{}, err
   303  	}
   304  
   305  	resp, err := c.sendReq(ctx, reqMeta, &sendOpt, endpoint)
   306  	if err != nil {
   307  		log.Error().Msg("the list of groups failed: " + err.Error())
   308  		return types.ListGroupsResult{}, err
   309  	}
   310  	defer utils.CloseResponse(resp)
   311  
   312  	// unmarshal the json content from response body
   313  	buf := new(strings.Builder)
   314  	_, err = io.Copy(buf, resp.Body)
   315  	if err != nil {
   316  		log.Error().Msg("the list of groups failed: " + err.Error())
   317  		return types.ListGroupsResult{}, err
   318  	}
   319  
   320  	listGroupsResult := types.ListGroupsResult{}
   321  	bufStr := buf.String()
   322  	err = json.Unmarshal([]byte(bufStr), &listGroupsResult)
   323  	if err != nil && listGroupsResult.Groups == nil {
   324  		log.Error().Msg("the list of groups failed: " + err.Error())
   325  		return types.ListGroupsResult{}, err
   326  	}
   327  
   328  	return listGroupsResult, nil
