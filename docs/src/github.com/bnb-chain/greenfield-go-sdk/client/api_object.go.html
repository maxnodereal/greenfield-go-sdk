   320  	reader io.Reader, opts types.PutObjectOptions,

   328  	if err != nil {
   329  		return err
   330  	}
   331  
   332  	// Total data read and written to server. should be equal to
   333  	// 'size' at the end of the call.
   334  	var totalUploadedSize int64
   335  
   336  	// Calculate the optimal parts info for a given size.
   337  	totalPartsCount, partSize, _, err := c.SplitPartInfo(objectSize, opts.PartSize)
   338  	if err != nil {
   339  		return err
   340  	}
   341  
   342  	// Part number always starts with '1'.
   343  	partNumber := 1
   344  	startPartNumber := int(offset/opts.PartSize + 1)
   345  
   346  	// Create a buffer.
   347  	buf := make([]byte, partSize)
   348  	complete := false
   349  
   350  	//  TODO(chris): Skip successful segments or add a verification file check.
   351  	for partNumber < startPartNumber {
   352  		length, rErr := utils.ReadFull(reader, buf)
   353  		if rErr == io.EOF && partNumber > 1 {
   354  			break
   355  		}
   356  		// Increment part number.
   357  		log.Debug().Msg(fmt.Sprintf("skip partNumber:%d, length:%d", partNumber, length))
   358  		// Save successfully uploaded size.
   359  		totalUploadedSize += int64(length)
   360  		partNumber++
   361  	}
   362  
   363  	for partNumber <= totalPartsCount {
   364  		if partNumber == totalPartsCount {
   365  			complete = true
   366  		}
   367  		if err = UploadSegmentHooker(partNumber); err != nil {
   368  			return err

   398  			bucketName:    bucketName,

   539  	if err == nil {
   540  		// If the destination exists and is a directory.
   541  		if st.IsDir() {
   542  			return errors.New("download file path is a directory")
   543  		}
   544  		return errors.New("download file already exist")
   545  	}
   546  
   547  	backoffDelay := types.DownloadBackOffDelay
   548  	var body io.ReadCloser
   549  	for retry := 0; retry < types.MaxDownloadTryTime; retry++ {
   550  		body, _, err = c.GetObject(ctx, bucketName, objectName, opts)
   551  		if err == nil {
   552  			break
   553  		}
   554  		if body != nil {
   555  			body.Close()
   556  		}
   557  
   558  		connectedFail := strings.Contains(err.Error(), types.GetConnectionFail)
   559  		if err != nil && !connectedFail {
   560  			return err
   561  		}
   562  
   563  		// connect the primary SP failed, try again
   564  		if opts.SupportRecovery {
   565  			// connect failed for 3 times, try to download piece from secondary SP
   566  			if retry == types.MaxDownloadTryTime-1 {
   567  				return c.RecoverObjectBySecondary(ctx, bucketName, objectName, filePath, opts)
   568  			}
   569  			continue
   570  		}
   571  
   572  		time.Sleep(backoffDelay)
   573  		backoffDelay *= 2
   574  	}
   575  
   576  	defer body.Close()
   577  
   578  	fd, err := os.OpenFile(filePath, os.O_CREATE|os.O_APPEND|os.O_WRONLY, 0o660)
   579  	if err != nil {
   580  		return err
   581  	}
   582  
   583  	_, err = io.Copy(fd, body)
   584  	fd.Close()
   585  	if err != nil {
   586  		return err
   587  	}
   588  
   589  	return nil
   590  }

   592  func (c *client) RecoverObjectBySecondary(ctx context.Context, bucketName, objectName, filePath string, opts types.GetObjectOptions) error {
   593  	// compute segment count
   594  	dataBlocks, parityBlocks, maxSegmentSize, err := c.GetRedundancyParams()
   595  	if err != nil {
   596  		return err
   597  	}
   598  	ecPieceCount := dataBlocks + parityBlocks
   599  	minRecoveryPieces := dataBlocks
   600  
   601  	objectInfo, err := c.HeadObject(ctx, bucketName, objectName)
   602  	if err != nil {
   603  		return err
   604  	}
   605  	objectSize := objectInfo.PayloadSize
   606  	startSegmentIdx, endSegmentIdx, diffStartOffset, diffEndOffset, err := checkGetObjectRange(objectSize, maxSegmentSize, opts)
   607  	if err != nil {
   608  		return err
   609  	}
   610  
   611  	secondaryEndpoints, err := c.getSecondaryEndpoints(ctx, objectInfo)
   612  	if err != nil {
   613  		return err
   614  	}
   615  
   616  	_, err = os.Stat(filePath)
   617  	if err == nil {
   618  		return errors.New("download file already exist:" + filePath)
   619  	}
   620  
   621  	f, err := os.OpenFile(filePath, os.O_APPEND|os.O_WRONLY|os.O_CREATE, 0o644)
   622  	if err != nil {
   623  		return err
   624  	}
   625  	defer f.Close()
   626  
   627  	// iterate over segment indices and recover one by one
   628  	for segmentIdx := startSegmentIdx; segmentIdx <= endSegmentIdx; segmentIdx++ {
   629  		doneCh := make(chan bool, len(secondaryEndpoints))
   630  		quitCh := make(chan bool)
   631  		totalTaskNum := int32(ecPieceCount)
   632  		doneTaskNum := uint32(0)
   633  		downLoadPieceSize := int64(0)
   634  		recoveryDataSources := make([][]byte, ecPieceCount)
   635  
   636  		segmentSize := utils.GetSegmentSize(objectSize, uint32(segmentIdx), maxSegmentSize)
   637  		ecPieceSize := utils.GetECPieceSize(objectSize, uint32(segmentIdx), maxSegmentSize, dataBlocks)
   638  		for ecIdx := 0; ecIdx < int(ecPieceCount); ecIdx++ {
   639  			recoveryDataSources[ecIdx] = nil
   640  			go func(secondaryIndex int) {
   641  				var responseBody io.ReadCloser
   642  				var pieceData []byte
   643  				defer func() {
   644  					// fmt.Printf("get done routine: %d \n", atomic.LoadUint32(&ecPieceCount))
   645  					// finish all the task, send signal to quitCh
   646  					if atomic.AddInt32(&totalTaskNum, -1) == 0 {
   647  						quitCh <- true
   648  					}
   649  					if responseBody != nil {
   650  						responseBody.Close()
   651  					}
   652  				}()
   653  				pieceInfo := types.QueryPieceInfo{
   654  					ObjectId:        strconv.FormatUint(objectInfo.Id.Uint64(), 10),
   655  					PieceIndex:      segmentIdx,
   656  					RedundancyIndex: secondaryIndex,
   657  				}
   658  				// call getSecondaryPieceData to retrieve recovery data for the segment
   659  				// TODO check if it is better with downloading data chunks first
   660  				responseBody, err = c.getSecondaryPieceData(ctx, bucketName, objectName, pieceInfo, types.GetSecondaryPieceOptions{Endpoint: secondaryEndpoints[secondaryIndex]})
   661  				if err == nil {
   662  					// convert recoveryData to byte
   663  					pieceData, err = io.ReadAll(responseBody)
   664  					if err != nil {
   665  						log.Error().Msg("read body err:" + err.Error())
   666  						return
   667  					}
   668  					pieceLen := int64(len(pieceData))
   669  					if pieceLen == ecPieceSize {
   670  						recoveryDataSources[secondaryIndex] = pieceData
   671  						doneCh <- true
   672  						atomic.StoreInt64(&downLoadPieceSize, pieceLen)
   673  					}
   674  				}
   675  			}(ecIdx)
   676  		}
   677  
   678  	loop:
   679  		for {
   680  			select {
   681  			case <-doneCh:
   682  				doneTaskNum++
   683  				// it is enough to recovery data with minRecoveryPieces EC data, no need to wait
   684  				if doneTaskNum >= minRecoveryPieces {
   685  					break loop
   686  				}
   687  			case <-quitCh: // all the task finish
   688  				if doneTaskNum < minRecoveryPieces { // finish task num not enough
   689  					return fmt.Errorf("get pieces from secondary not enough %d", doneTaskNum)
   690  				}
   691  				ecTotalSize := int64(uint32(downLoadPieceSize) * dataBlocks)
   692  				if ecTotalSize < segmentSize || ecTotalSize > segmentSize+4 {
   693  					return fmt.Errorf("get secondary piece data length error ecChunks total size:%d, segment size:%d", ecTotalSize, segmentSize)
   694  				}
   695  			}
   696  		}
   697  
   698  		// decode the original segment data from the piece data
   699  		recoverySegData, err := redundancy.DecodeRawSegment(recoveryDataSources, segmentSize, int(dataBlocks), int(parityBlocks))
   700  		if err != nil {
   701  			return fmt.Errorf("decode segment err when recovery, segment id:%d err: %s", segmentIdx, err.Error())
   702  		}
   703  
   704  		isFirstSeg := segmentIdx == startSegmentIdx && diffStartOffset > 0
   705  		isLastSeg := segmentIdx == endSegmentIdx && diffEndOffset > 0
   706  
   707  		// deal with range recovery
   708  		if isFirstSeg && isLastSeg {
   709  			recoverySegData = recoverySegData[diffStartOffset : segmentSize-diffEndOffset]
   710  		} else if isFirstSeg {
   711  			recoverySegData = recoverySegData[diffStartOffset:]
   712  		} else if isLastSeg {
   713  			recoverySegData = recoverySegData[:segmentSize-diffEndOffset]
   714  		}
   715  
   716  		_, err = f.Write(recoverySegData)
   717  		if err != nil {
   718  			return err
   719  		}
   720  	}
   721  
   722  	return nil
   723  }
