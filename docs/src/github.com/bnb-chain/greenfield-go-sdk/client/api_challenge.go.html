    33  // The challenge info includes the piece data, piece hash roots and integrity hash corresponding to the accessed SP
    34  func (c *client) GetChallengeInfo(ctx context.Context, objectID string, pieceIndex, redundancyIndex int, opts types.GetChallengeInfoOptions) (types.ChallengeResult, error) {
    35  	if objectID == "" {
    36  		return types.ChallengeResult{}, errors.New("fail to get objectId")
    37  	}
    38  
    39  	if pieceIndex < 0 {
    40  		return types.ChallengeResult{}, errors.New("index error, should be 0 to parityShards plus dataShards")
    41  	}
    42  
    43  	var err error
    44  	dataBlocks, parityBlocks, _, err := c.GetRedundancyParams()
    45  	if err != nil {
    46  		return types.ChallengeResult{}, errors.New("fail to get redundancy params:" + err.Error())
    47  	}
    48  	maxRedundancyIndex := int(dataBlocks+parityBlocks) - 1
    49  	if redundancyIndex < types.PrimaryRedundancyIndex || redundancyIndex > maxRedundancyIndex {
    50  		return types.ChallengeResult{}, fmt.Errorf("redundancy index invalid, the index should be %d to %d", types.PrimaryRedundancyIndex, maxRedundancyIndex)
    51  	}
    52  
    53  	reqMeta := requestMeta{
    54  		urlRelPath:    types.ChallengeUrl,
    55  		contentSHA256: types.EmptyStringSHA256,
    56  		pieceInfo: types.QueryPieceInfo{
    57  			ObjectId:        objectID,
    58  			PieceIndex:      pieceIndex,
    59  			RedundancyIndex: redundancyIndex,
    60  		},
    61  	}
    62  
    63  	sendOpt := sendOptions{
    64  		method:           http.MethodGet,
    65  		isAdminApi:       true,
    66  		disableCloseBody: true,
    67  	}
    68  
    69  	var endpoint *url.URL
    70  	if opts.Endpoint != "" {
    71  		var useHttps bool
    72  		if strings.Contains(opts.Endpoint, "https") {
    73  			useHttps = true
    74  		} else {
    75  			useHttps = c.secure
    76  		}
    77  
    78  		endpoint, err = utils.GetEndpointURL(opts.Endpoint, useHttps)
    79  		if err != nil {
    80  			log.Error().Msg(fmt.Sprintf("fetch endpoint from opts %s fail:%v", opts.Endpoint, err))
    81  			return types.ChallengeResult{}, err
    82  		}
    83  	} else if opts.SPAddress != "" {
    84  		// get endpoint from sp address
    85  		endpoint, err = c.getSPUrlByAddr(opts.SPAddress)
    86  		if err != nil {
    87  			log.Error().Msg(fmt.Sprintf("route endpoint by sp address: %s failed, err: %v", opts.SPAddress, err))
    88  			return types.ChallengeResult{}, err
    89  		}
    90  	} else {
    91  		// get sp address info based on the redundancy index
    92  		objectInfo, err := c.HeadObjectByID(ctx, objectID)
    93  		if err != nil {
    94  			return types.ChallengeResult{}, err
    95  		}
    96  
    97  		if redundancyIndex == types.PrimaryRedundancyIndex {
    98  			// get endpoint of primary sp
    99  			endpoint, err = c.getSPUrlByBucket(objectInfo.BucketName)
   100  			if err != nil {
   101  				log.Error().Msg(fmt.Sprintf("route endpoint by bucket: %s failed, err: %v", objectInfo.BucketName, err))
   102  				return types.ChallengeResult{}, err
   103  			}
   104  		} else {
   105  			// get endpoint of the secondary sp
   106  			secondarySP := objectInfo.SecondarySpAddresses[redundancyIndex]
   107  			endpoint, err = c.getSPUrlByAddr(secondarySP)
   108  			if err != nil {
   109  				log.Error().Msg(fmt.Sprintf("route endpoint by sp address: %s failed, err: %v", secondarySP, err))
   110  				return types.ChallengeResult{}, err
   111  			}
   112  		}
   113  	}
   114  
   115  	resp, err := c.sendReq(ctx, reqMeta, &sendOpt, endpoint)
   116  	if err != nil {
   117  		return types.ChallengeResult{}, err
   118  	}
   119  
   120  	// fetch integrity hash
   121  	integrityHash := resp.Header.Get(types.HTTPHeaderIntegrityHash)
   122  	// fetch piece hashes
   123  	pieceHashes := resp.Header.Get(types.HTTPHeaderPieceHash)
   124  
   125  	if integrityHash == "" || pieceHashes == "" {
   126  		utils.CloseResponse(resp)
   127  		return types.ChallengeResult{}, errors.New("fail to fetch hash info")
   128  	}
   129  
   130  	hashList := strings.Split(pieceHashes, ",")
   131  	// min hash num equals one segment hash plus EC dataShards, parityShards
   132  	if len(hashList) < 1 {
   133  		return types.ChallengeResult{}, errors.New("get piece hashes less than 1")
   134  	}
   135  
   136  	result := types.ChallengeResult{
   137  		PieceData:     resp.Body,
   138  		IntegrityHash: integrityHash,
   139  		PiecesHash:    hashList,
   140  	}
   141  	return result, nil
   142  }
