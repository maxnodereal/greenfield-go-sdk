   243  // newRequest constructs the http request, set url, body and headers
   244  func (c *client) newRequest(ctx context.Context, method string, meta requestMeta,
   245  	body interface{}, txnHash string, isAdminAPi bool, endpoint *url.URL,
   246  ) (req *http.Request, err error) {
   247  	isVirtualHost := c.isVirtualHostStyleUrl(*endpoint, meta.bucketName)
   248  	// construct the target url
   249  	desURL, err := c.generateURL(meta.bucketName, meta.objectName, meta.urlRelPath,
   250  		meta.urlValues, isAdminAPi, endpoint, isVirtualHost)
   251  	if err != nil {
   252  		log.Error().Msg(fmt.Sprintf("generate request url on SP: %s fail, err: %s", endpoint.String(), err))
   253  		return nil, err
   254  	}
   255  
   256  	var reader io.Reader
   257  	contentType := ""
   258  	sha256Hex := ""
   259  	if body != nil {
   260  		// the body content is io.Reader type
   261  		if ObjectReader, ok := body.(io.Reader); ok {
   262  			reader = ObjectReader
   263  			if meta.contentType == "" {
   264  				contentType = types.ContentDefault
   265  			}
   266  		} else {
   267  			// the body content is xml type
   268  			content, err := xml.Marshal(body)
   269  			if err != nil {
   270  				return nil, err
   271  			}
   272  			contentType = types.ContentTypeXML
   273  			reader = bytes.NewReader(content)
   274  			sha256Hex = utils.CalcSHA256Hex(content)
   275  		}
   276  	}
   277  
   278  	// Initialize a new HTTP request for the method.
   279  	req, err = http.NewRequestWithContext(ctx, method, desURL.String(), nil)
   280  	if err != nil {
   281  		return nil, err
   282  	}
   283  
   284  	// need to turn the body into ReadCloser
   285  	if body == nil {
   286  		req.Body = nil
   287  	} else {
   288  		req.Body = io.NopCloser(reader)
   289  	}
   290  
   291  	// set content length
   292  	req.ContentLength = meta.contentLength
   293  
   294  	// set txn hash header
   295  	if txnHash != "" {
   296  		req.Header.Set(types.HTTPHeaderTransactionHash, txnHash)
   297  	}
   298  
   299  	// set content type header
   300  	if meta.contentType != "" {
   301  		req.Header.Set(types.HTTPHeaderContentType, meta.contentType)
   302  	} else if contentType != "" {
   303  		req.Header.Set(types.HTTPHeaderContentType, contentType)
   304  	} else {
   305  		req.Header.Set(types.HTTPHeaderContentType, types.ContentDefault)
   306  	}
   307  
   308  	// set md5 header
   309  	if meta.contentMD5Base64 != "" {
   310  		req.Header[types.HTTPHeaderContentMD5] = []string{meta.contentMD5Base64}
   311  	}
   312  
   313  	// set sha256 header
   314  	if meta.contentSHA256 != "" {
   315  		req.Header[types.HTTPHeaderContentSHA256] = []string{meta.contentSHA256}
   316  	} else {
   317  		req.Header[types.HTTPHeaderContentSHA256] = []string{sha256Hex}
   318  	}
   319  
   320  	if meta.rangeInfo != "" && method == http.MethodGet {
   321  		req.Header.Set(types.HTTPHeaderRange, meta.rangeInfo)
   322  	}
   323  
   324  	// if pieceInfo.ObjectId is not empty, other field should be set as well
   325  	if meta.pieceInfo.ObjectId != "" {
   326  		info := meta.pieceInfo
   327  		req.Header.Set(types.HTTPHeaderObjectID, info.ObjectId)
   328  		req.Header.Set(types.HTTPHeaderRedundancyIndex, strconv.Itoa(info.RedundancyIndex))
   329  		req.Header.Set(types.HTTPHeaderPieceIndex, strconv.Itoa(info.PieceIndex))
   330  	}
   331  
   332  	if isAdminAPi {
   333  		if meta.txnMsg != "" {
   334  			req.Header.Set(types.HTTPHeaderUnsignedMsg, meta.txnMsg)
   335  		}
   336  	} else {
   337  		// set request host
   338  		if c.host != "" {
   339  			req.Host = c.host
   340  		} else if req.URL.Host != "" {
   341  			req.Host = req.URL.Host
   342  		}
   343  	}
   344  
   345  	if meta.userAddress != "" {
   346  		req.Header.Set(types.HTTPHeaderUserAddress, meta.userAddress)
   347  	}
   348  
   349  	// set date header
   350  	stNow := time.Now().UTC()
   351  	req.Header.Set(types.HTTPHeaderDate, stNow.Format(types.Iso8601DateFormatSecond))
   352  
   353  	// set user-agent
   354  	req.Header.Set(types.HTTPHeaderUserAgent, c.userAgent)
   355  
   356  	// sign the total http request info when auth type v1
   357  	err = c.signRequest(req)
   358  	if err != nil {
   359  		return req, err
   360  	}
   361  
   362  	return
   363  }

   365  // doAPI call client.Do() to send request and read response from servers
   366  func (c *client) doAPI(ctx context.Context, req *http.Request, meta requestMeta, closeBody bool) (*http.Response, error) {
   367  	var cancel context.CancelFunc
   368  	if closeBody {
   369  		ctx, cancel = context.WithCancel(ctx)
   370  		defer cancel()
   371  	}
   372  	req = req.WithContext(ctx)
   373  
   374  	resp, err := c.httpClient.Do(req)
   375  	if err != nil {
   376  		// If we got an error, and the context has been canceled,
   377  		// the context's error is probably more useful.
   378  		select {
   379  		case <-ctx.Done():
   380  			return nil, ctx.Err()
   381  		default:
   382  		}
   383  		if urlErr, ok := err.(*url.Error); ok {
   384  			if strings.Contains(urlErr.Err.Error(), "EOF") {
   385  				return nil, &url.Error{
   386  					Op:  urlErr.Op,
   387  					URL: urlErr.URL,
   388  					Err: errors.New("Connection closed by foreign host " + urlErr.URL + ". Retry again."),
   389  				}
   390  			}
   391  		}
   392  		return nil, err
   393  	}
   394  	defer func() {
   395  		if closeBody {
   396  			utils.CloseResponse(resp)
   397  		}
   398  	}()
   399  
   400  	// construct err responses and messages
   401  	err = types.ConstructErrResponse(resp, meta.bucketName, meta.objectName)
   402  	if err != nil {
   403  		// dump error msg
   404  		if c.isTraceEnabled {
   405  			c.dumpSPMsg(req, resp)
   406  		}
   407  		return resp, err
   408  	}
   409  
   410  	// dump msg
   411  	if c.isTraceEnabled && !c.onlyTraceError {
   412  		c.dumpSPMsg(req, resp)
   413  	}
   414  
   415  	return resp, nil
   416  }

   444  // generateURL constructs the target request url based on the parameters
   445  func (c *client) generateURL(bucketName string, objectName string, relativePath string,
   446  	queryValues url.Values, isAdminApi bool, endpoint *url.URL, isVirtualHost bool,
   447  ) (*url.URL, error) {
   448  	host := endpoint.Host
   449  	scheme := endpoint.Scheme
   450  
   451  	// Strip port 80 and 443
   452  	if h, p, err := net.SplitHostPort(host); err == nil {
   453  		if scheme == "http" && p == "80" || scheme == "https" && p == "443" {
   454  			host = h
   455  			if ip := net.ParseIP(h); ip != nil && ip.To16() != nil {
   456  				host = "[" + h + "]"
   457  			}
   458  		}
   459  	}
   460  
   461  	var urlStr string
   462  	if isAdminApi {
   463  		prefix := types.AdminURLPrefix + types.AdminURLVersion
   464  		urlStr = scheme + "://" + host + prefix + "/"
   465  	} else {
   466  		urlStr = scheme + "://" + host + "/"
   467  		if bucketName != "" {
   468  			if isVirtualHost {
   469  				// set virtual host url
   470  				urlStr = scheme + "://" + bucketName + "." + host + "/"
   471  			} else {
   472  				// set path style url
   473  				urlStr = urlStr + bucketName + "/"
   474  			}
   475  
   476  			if objectName != "" {
   477  				urlStr += utils.EncodePath(objectName)
   478  			}
   479  		}
   480  	}
   481  
   482  	if relativePath != "" {
   483  		urlStr += utils.EncodePath(relativePath)
   484  	}
   485  
   486  	if len(queryValues) > 0 {
   487  		urlStrNew, err := utils.AddQueryValues(urlStr, queryValues)
   488  		if err != nil {
   489  			return nil, err
   490  		}
   491  		urlStr = urlStrNew
   492  	}
   493  
   494  	return url.Parse(urlStr)
   495  }
