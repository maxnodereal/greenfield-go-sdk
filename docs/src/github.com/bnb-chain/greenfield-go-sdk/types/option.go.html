   212  func (o *GetObjectOptions) SetRange(start, end int64) error {
   213  	switch {
   214  	case 0 < start && end == 0:
   215  		// `bytes=N-`.
   216  		o.Range = fmt.Sprintf("bytes=%d-", start)
   217  	case 0 <= start && start <= end:
   218  		// `bytes=N-M`
   219  		o.Range = fmt.Sprintf("bytes=%d-%d", start, end)
   220  	default:
   221  		return ToInvalidArgumentResp(
   222  			fmt.Sprintf(
   223  				"Invalid Range : start=%d end=%d",
   224  				start, end))

   226  	return nil
