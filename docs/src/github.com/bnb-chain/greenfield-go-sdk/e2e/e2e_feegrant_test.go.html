    21  func (s *FeeGrantTestSuite) Test_FeeGrant() {
    22  	cli := s.Client
    23  	t := s.T()
    24  	ctx := s.ClientContext
    25  
    26  	granter, _, err := types.NewAccount("granter")
    27  	s.Require().NoError(err)
    28  	granterAddr := granter.GetAddress().String()
    29  	grantee, _, err := types.NewAccount("grantee")
    30  	s.Require().NoError(err)
    31  	granteeAddr := grantee.GetAddress().String()
    32  
    33  	// charge granter and grantee accounts
    34  	chargeAmount := math.NewIntWithDecimal(1, 19)
    35  	transferDetails := make([]types.TransferDetail, 0)
    36  	transferDetails = append(transferDetails, types.TransferDetail{
    37  		ToAddress: granterAddr,
    38  		Amount:    chargeAmount,
    39  	})
    40  	transferDetails = append(transferDetails, types.TransferDetail{
    41  		ToAddress: granteeAddr,
    42  		Amount:    chargeAmount,
    43  	})
    44  
    45  	txHash, err := cli.MultiTransfer(ctx, transferDetails, types2.TxOption{})
    46  	s.Require().NoError(err)
    47  	_, err = cli.WaitForTx(s.ClientContext, txHash)
    48  	s.Require().NoError(err)
    49  
    50  	// granter grants allowance to grantee
    51  	cli.SetDefaultAccount(granter)
    52  	allowanceAmount := math.NewIntWithDecimal(1, 18)
    53  
    54  	txHash, err = cli.GrantBasicAllowance(ctx, granteeAddr, allowanceAmount, nil, types2.TxOption{})
    55  	s.Require().NoError(err)
    56  	_, err = cli.WaitForTx(ctx, txHash)
    57  	s.Require().NoError(err)
    58  
    59  	// Query the allowance
    60  	allowance, err := cli.QueryBasicAllowance(ctx, granterAddr, granteeAddr)
    61  	s.Require().NoError(err)
    62  	t.Log(allowance.String())
    63  
    64  	// show grantee balance before the grantee making a tx
    65  	granteeBalanceBefore, err := cli.GetAccountBalance(ctx, granteeAddr)
    66  	s.Require().NoError(err)
    67  
    68  	// grantee makes a tx and costs the fee provided by granter
    69  	cli.SetDefaultAccount(grantee)
    70  	txHash, err = cli.CreatePaymentAccount(ctx, granteeAddr, types2.TxOption{
    71  		FeeGranter: granter.GetAddress(),
    72  	})
    73  
    74  	s.Require().NoError(err)
    75  	_, _ = cli.WaitForTx(ctx, txHash)
    76  
    77  	granteeBalanceAfter, err := cli.GetAccountBalance(ctx, granteeAddr)
    78  	s.Require().NoError(err)
    79  
    80  	// grantee balance stays still
    81  	s.Require().Equal(granteeBalanceBefore, granteeBalanceAfter)
    82  
    83  	// the granter revokes
    84  	cli.SetDefaultAccount(granter)
    85  	txHash, err = cli.RevokeAllowance(ctx, granteeAddr, types2.TxOption{})
    86  	s.Require().NoError(err)
    87  	_, _ = cli.WaitForTx(ctx, txHash)
    88  
    89  	// Query the allowance
    90  	_, err = cli.QueryBasicAllowance(ctx, granterAddr, granteeAddr)
    91  	s.Require().Error(err)
    92  
    93  	// transaction is failed
    94  	cli.SetDefaultAccount(grantee)
    95  	_, err = cli.CreatePaymentAccount(ctx, granteeAddr, types2.TxOption{
    96  		FeeGranter: granter.GetAddress(),
    97  	})
    98  	s.Require().Error(err)
    99  }
