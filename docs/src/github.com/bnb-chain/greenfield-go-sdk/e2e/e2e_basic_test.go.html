   138  func (s *BasicTestSuite) Test_Payment() {
   139  	account := s.DefaultAccount
   140  	cli := s.Client
   141  	t := s.T()
   142  	ctx := s.ClientContext
   143  
   144  	paymentAccountsBeforeCreate, err := cli.GetPaymentAccountsByOwner(ctx, account.GetAddress().String())
   145  	s.Require().NoError(err)
   146  	txHash, err := cli.CreatePaymentAccount(ctx, account.GetAddress().String(), types2.TxOption{})
   147  	s.Require().NoError(err)
   148  	t.Logf("Acc: %s", txHash)
   149  	waitForTx, err := cli.WaitForTx(ctx, txHash)
   150  	s.Require().NoError(err)
   151  	t.Logf("Wair for tx: %s", waitForTx.String())
   152  
   153  	paymentAccountsByOwnerAfterCreate, err := cli.GetPaymentAccountsByOwner(ctx, account.GetAddress().String())
   154  	s.Require().NoError(err)
   155  	s.Require().Equal(len(paymentAccountsByOwnerAfterCreate)-len(paymentAccountsBeforeCreate), 1)
   156  
   157  	// deposit
   158  	paymentAddr := paymentAccountsByOwnerAfterCreate[len(paymentAccountsByOwnerAfterCreate)-1].Addr
   159  	depositAmount := math.NewIntFromUint64(100)
   160  	depositTxHash, err := cli.Deposit(ctx, paymentAddr, depositAmount, types2.TxOption{})
   161  	s.Require().NoError(err)
   162  	t.Logf("deposit tx: %s", depositTxHash)
   163  	waitForTx, err = cli.WaitForTx(ctx, depositTxHash)
   164  	s.Require().NoError(err)
   165  	t.Logf("Wair for tx: %s", waitForTx.String())
   166  
   167  	// get stream record
   168  	streamRecord, err := cli.GetStreamRecord(ctx, paymentAddr)
   169  	s.Require().NoError(err)
   170  	s.Require().Equal(streamRecord.StaticBalance.String(), depositAmount.String())
   171  
   172  	// withdraw
   173  	withdrawAmount := math.NewIntFromUint64(50)
   174  	withdrawTxHash, err := cli.Withdraw(ctx, paymentAddr, withdrawAmount, types2.TxOption{})
   175  	s.Require().NoError(err)
   176  	t.Logf("withdraw tx: %s", withdrawTxHash)
   177  	waitForTx, err = cli.WaitForTx(ctx, withdrawTxHash)
   178  	s.Require().NoError(err)
   179  	t.Logf("Wair for tx: %s", waitForTx.String())
   180  	streamRecordAfterWithdraw, err := cli.GetStreamRecord(ctx, paymentAddr)
   181  	s.Require().NoError(err)
   182  	s.Require().Equal(streamRecordAfterWithdraw.StaticBalance.String(), depositAmount.Sub(withdrawAmount).String())
   183  
   184  	// disable refund
   185  	paymentAccountBeforeDisableRefund, err := cli.GetPaymentAccount(ctx, paymentAddr)
   186  	s.Require().NoError(err)
   187  	s.Require().True(paymentAccountBeforeDisableRefund.Refundable)
   188  	disableRefundTxHash, err := cli.DisableRefund(ctx, paymentAddr, types2.TxOption{})
   189  	s.Require().NoError(err)
   190  	t.Logf("disable refund tx: %s", disableRefundTxHash)
   191  	waitForTx, err = cli.WaitForTx(ctx, disableRefundTxHash)
   192  	s.Require().NoError(err)
   193  	t.Logf("Wair for tx: %s", waitForTx.String())
   194  	paymentAccountAfterDisableRefund, err := cli.GetPaymentAccount(ctx, paymentAddr)
   195  	s.Require().NoError(err)
   196  	s.Require().False(paymentAccountAfterDisableRefund.Refundable)
   197  }
